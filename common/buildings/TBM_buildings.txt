building_xeno_synaptic_node = {
	
	can_build = yes
	can_demolish = yes
	can_be_ruined = yes
	can_be_disabled = yes

	position_priority = 2

	category = research

	building_sets = {
		research
	}

	potential = {
		is_starting_planet = no
		NOT = { is_planet_class = pc_habitat }
		exists = owner
		owner = { 
			is_gestalt = yes
			is_fallen_empire = no
		}
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			uses_district_set = cosmogenesis_world
		}
	}

	abort_trigger = {
		exists = owner
		OR = {
			has_major_upgraded_capital = yes
			owner = { 
				OR = {
					is_gestalt = no
					is_fallen_empire = no
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_gestalt = no }
			has_major_upgraded_capital = yes
		}
	}

	resources = {
		category = planet_buildings
		upkeep = {
			energy = 10		# Who said thinking hard was easy?
		}
		cost = {
			minerals = 600
		}
		produces = {		
			physics_research = 1
			multiplier = value:feature_research_gestalt_phys
		}
		produces = {
			society_research = 1
			multiplier = value:feature_research_gestalt_soc
		}
		produces = {
			engineering_research = 1
			multiplier = value:feature_research_gestalt_eng
		}
	}
}
