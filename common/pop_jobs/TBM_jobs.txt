### The Blocker Mod - Custon Jobs
#	No vanilla files are changed here.


# Restores Rangers from pre-4.0. TBM makes Rangers into colonist-style jobs, allowing Environmentalist empires to always make use of the significant Colonist bonuses in perpetuity, in exchange for being unable to clear blockers. This also allows the AI to cope much better and saves updating ~1000 lines of vanilla code every time there's an update to deposit calcs for Rangers.
ranger = {
	category = specialist

	tags = {
		colonist
	}

	swappable_data = {
		default = {
			condition_string = SPECIALIST_JOB_TRIGGER
			building_icon = building_research_lab_1
		}

		swap_type = {
			trigger = {
				always = yes
			}
			name = ranger
			desc = ranger
			icon = ranger
			building_icon = holding_park_ranger_lodge
			weight = 5
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_specialist_job
	possible = {
		complex_specialist_job_check_trigger = yes
	}

	resources = {
		category = planet_colonist		# This line is important! It lets Rangers get bonuses from the parent category.
		produces = {
			society_research = 1
		}
	}

}

dangerous_wildlife_hunter = {
	category = worker
	clothes_texture_index = 2

	swappable_data = {
		default = { 
			building_icon = building_food_processing_facility
			icon = farmer
		}
	}

	tags = {
		colonist
	}

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_worker_job
	possible = {
		hidden_trigger = { exists = owner }
		owner = { is_gestalt = no }
	}

	
	resources = {
		category = planet_farmers
		produces = {
			food = 4
			unity = 1
		}
		upkeep = {
			energy = 1
		}
	}
	
	overlord_resources = {
		category = planet_requisitioned_food
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_produce_requisitorium
				}
			}
			food = @requisitorium_food_overlord
		}
	}

	weight = {
		weight = @worker_job_weight
		
		# low income modifier
		modifier = {
			factor = 1.2
			exists = planet
			planet = {
				has_available_jobs = farmer
			}
			exists = owner
			owner = {
				is_ai = yes
				has_monthly_income = {
					resource = food
					value < 10
				}
			}
		}

		modifier = {
			factor = 0.5
			has_trait = trait_lithoid

			owner = {
				is_lithoid_empire = yes
				has_monthly_income = {
					resource = food
					value >= 0
				}
			}
		}
	}
}
