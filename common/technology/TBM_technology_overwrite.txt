### The Blocker Mod - Overwrite - Blocker Techs
#	This overwrite changes the tech weights to draw without needing particular blockers.

tech_mountain_range = {
	cost = @tier2cost2
	area = society
	tier = 2
	category = { new_worlds }
	prerequisites = { "tech_colonization_1" }
	weight = @tier2weight2

	weight_groups = {
		deposit_blockers
	}
	mod_weight_if_group_picked = {
		deposit_blockers = 0.5
	}

	weight_modifier = {
		modifier = {
			factor = 2
		}
		modifier = {
			factor = 1.25
			has_tradition = tr_expansion_adopt
		}
		modifier = {
			factor = 4
			has_ascension_perk = ap_mastery_of_nature
		}
		# modifier = {
		# 	factor = 0
		# 	NOT = {
		# 		any_owned_planet = {
		# 			has_deposit = "d_mountain_range"
		# 		}
		# 	}
		# }

		modifier = {
			factor = 0
			num_owned_planets < 2
		}
	}

	ai_weight = {
		modifier = {
			factor = 3
		}
		modifier = {
			factor = 0
			NOT = {
				any_owned_planet = {
					has_deposit = "d_mountain_range"
				}
			}
		}

		modifier = {
			factor = 0
			num_owned_planets < 2
		}
	}
}
