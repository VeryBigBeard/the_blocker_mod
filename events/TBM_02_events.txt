		name = tbm.51.kelp

		}

			limit = {
				any_owned_pop_group = {
					has_trait = trait_aquatic

			}

				limit = {
					has_trait = trait_aquatic
				}		
				kill_pop_group = {
					pop_group = from
					amount = 40
				}
			}
			random_owned_pop_group = {
				limit = {
					has_trait = trait_aquatic
				}		
				kill_pop_group = {
					pop_group = from
					amount = 60
				}
			}
		}
	}

# 53. Fish Attack
planet_event = {
	id = tbm.53

	picture = GFX_evt_unspeakable_horror
	show_sound = dragon_attack_mouth

	hide_window = yes
	is_triggered_only = yes

	pre_triggers = {
		has_owner = yes
		is_homeworld = no
	}

	trigger = {
		is_starting_planet = no
		has_deposit = d_submerged_ore_veins
		has_district = district_mining
	}

	# Handler
	immediate = {

		this = {
			save_event_target_as = fish_planet
		}

		random_list = {
			5 = {
				modifier = {
					add = 5
					is_wet = yes
				}
				modifier = {
					factor = 2
					owner_main_species = {
						has_trait = trait_aquatic
					}
				}
				modifier = {
					add = 10
					is_planet_class = pc_ocean
				}
				modifier = {
					add = 10
					OR = {
						has_planet_flag = planet_drenched	# Something unexpected got in there.
						check_variable = {
							which = ice_dumping_happened
							value > 0
						}
					}
				}
				modifier = {
					factor = 0.25
					has_planet_flag = docile_fish
				}

				# Set whether crew will escape or not.
				random_list = {
					50 = {
						modifier = {
							factor = 2
							owner_main_species = { has_trait = trait_aquatic	}
						}
						set_planet_flag = crew_escaped
					}
					50 = {}
				}

				# Fish Attack!
				planet_event = { id = tbm.54 days = 1 }
			}
			85 = {}
		}
	}
}

	immediate = {

			limit = {
				num_free_districts = {

				}
			}

			type = "SUDDEN_SINKHOLE"
			localization = tbm_message_sudden_sinkhole
			days = 30
			target = this
		}
	}
}

### 84. Vicious Sandstorm (Handler) (Various)
planet_event = {
	id = tbm.84

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		exists = owner
		owner = { is_country_type = default }
		is_dry = yes
		OR = {
			has_deposit = d_searing_desert
			has_deposit = d_sandsea
			has_deposit = d_great_desert
			has_deposit = d_great_city
		}
		NOT = { has_planet_flag = sandstorm }
	}

	immediate = {
		hidden_effect = {		
			set_variable = {
				which = @trees
				value = value:forestation
			}
		}
		
		random_list = {
			5 = {

								is_robotic_species = yes
							}
							count > 0
						}
					}
					is_gestalt = yes
				}
			}	
		}
	}
	desc = {
		text = tbm.90_desc_miner_deaths
		trigger = {
			check_variable = {
				which = @casualty_chance
				value > 50
			}
		}
	}
	desc = {
		text = tbm.90_desc_no_deaths
		trigger = {
			check_variable = {
				which = @casualty_chance
				value <= 50
			}
		}
	}
	picture = GFX_evt_collapsing_roof
	show_sound = event_structural_collapse

	option = {
		name = UNFORTUNATE
	}

	after = {

		### Damage
		event_target:collapsed_mine = {
			remove_deposit = yes
		}
		
		### Casualties
		if = {
			limit = {
				check_variable = {
					which = @casualty_chance
					value > 50
				}
			}
			if = {
				limit = {
					owner = {
						OR = {
							AND = {
								has_policy_flag = ai_servitude
								count_owned_pop_amount = {

										is_robotic_species = yes

							}

						}
					}
				}

					limit = {
						OR = {
							AND = {
								is_robotic_species = yes

							}

						}
					}

						pop_group = from
						amount = 50
					}
				}
			}
			else = {
				random_owned_pop_group = {
					kill_pop_group = {
